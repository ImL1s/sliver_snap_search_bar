```dart
@override
void didUpdateWidget(covariant SnapSearchBarView old) {
  super.didUpdateWidget(old);

  final scrollChanged = old.scrollController != widget.scrollController;
  final configChanged =
      old.totalHeight != widget.totalHeight ||
      old.snapDuration != widget.snapDuration ||
      old.snapCurve != widget.snapCurve;

  if (scrollChanged) {
    if (_ownsScrollCtrl) _scrollCtrl.dispose();
    _scrollCtrl = widget.scrollController ?? ScrollController();
    _ownsScrollCtrl = widget.scrollController == null;
  }

  if (scrollChanged || configChanged) {
    _snapCtrl.dispose();
    _snapCtrl = SnapSearchBarController(
      scrollController: _scrollCtrl,
      totalHeight: widget.totalHeight,
      snapDuration: widget.snapDuration,
      snapCurve: widget.snapCurve,
    );
  }

  if (old.isSearching != widget.isSearching) {
    if (widget.isSearching) {
      _snapCtrl.savePreSearchOffset();
      WidgetsBinding.instance.addPostFrameCallback((_) {
        if (!mounted || !_scrollCtrl.hasClients) return;
        if (_scrollCtrl.offset != 0) _scrollCtrl.jumpTo(0);
      });
    } else {
      _snapCtrl.restorePreSearchOffset();
    }
  }
}
```

## Task 4: [Codex-3] `restorePreSearchOffset` retries on `!hasClients`

**Why:** On tab detach/reattach or first frame of a fresh `CustomScrollView`, `hasClients` can be false briefly. Current code bails silently ‚Üí saved offset lost.

**Files:**
- Modify: `lib/src/sliver_snap_controller.dart` `_restoreInternal` (~lines 180-200)
- Test: `test/sliver_snap_search_bar_test.dart` SnapSearchBarController group

**Step 1: Write the failing test**

```dart
test('restore retries when hasClients=false on first frame', () async {
  // Start with an orphan controller (no clients) and confirm restore
  // does not silently abandon the saved offset. After attaching in a
  // subsequent frame, the expected jumpTo should still run.
  final scroll = ScrollController();
  final ctl = SnapSearchBarController(scrollController: scroll);
  ctl.savePreSearchOffset(); // saves 0 safely with no clients
  ctl.restorePreSearchOffset();
  // Without retry, restoreExhausted would never fire. With retry and
  // no attachment, it fires after maxRestoreAttempts.
  // Smoke: no exception thrown; exhaustion path exercised.
  // Deeper: capture via onRestoreExhausted.
  scroll.dispose();
  ctl.dispose();
});
```

**Step 2: Run**: confirm no exception.

**Step 3: Implement**

In `sliver_snap_controller.dart` `_restoreInternal`:

```dart
void _restoreInternal(int version, [int attempt = 0]) {
  WidgetsBinding.instance.addPostFrameCallback((_) {
    if (_disposed || version != _version) return;
    // Round 2 fix (Codex #3): hasClients may be false on the first
    // post-frame after detach/re-attach. Retry instead of bailing.
    if (!scrollController.hasClients) {
      if (attempt < maxRestoreAttempts) {
        _restoreInternal(version, attempt + 1);
      } else {
        onRestoreExhausted?.call(_preSearchOffset);
      }
      return;
    }
    final position = scrollController.position;
    if (!position.hasContentDimensions) {
      if (attempt < maxRestoreAttempts) {
        _restoreInternal(version, attempt + 1);
      } else {
        onRestoreExhausted?.call(_preSearchOffset);
        scrollController.jumpTo(0);
      }
      return;
    }
    final safeOffset = _preSearchOffset.clamp(
      position.minScrollExtent,
      position.maxScrollExtent,
    );
    scrollController.jumpTo(safeOffset < totalHeight ? 0.0 : safeOffset);
  });
}
```
